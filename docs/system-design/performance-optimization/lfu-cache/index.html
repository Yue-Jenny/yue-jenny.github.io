class DLinkedList {
    constructor() {
        // å»ºç«‹å…©å€‹ nodeï¼Œåˆ†åˆ¥ç‚º head èˆ‡ tail

        node.next.prev = node.prev;

    addNode(node) {

        node.next = this.tail;
        // node çš„ prev æŒ‡æ¨™æŒ‡å‘ tail çš„å‰ä¸€å€‹ node (tail.prev)

        this.tail.prev.next = node;
        // tail çš„ prev æŒ‡æ¨™æŒ‡å‘ node
        this.tail.prev = node;
    }

    removeHead() {
        let node = this.head.next;
        this.removeNode(node);
        return node.key;
    }

    isEmpty() {
        return this.head.next === this.tail;
    }
}

class LFUCache {
    constructor(capacity) {
        this.capacity = capacity;
        this.currentSize = 0;
        this.leastFrequency = 0;
        this.cache = new Map();  // store { key : Node }
        this.frequencyList = new Map();  // store { frequency : DLinkedList }
    }

    get(key) {
        if (!this.cache.has(key)) {
            return -1;
        }
        let node = this.cache.get(key);
        this.updateFrequency(node);
        return node.value;
    }

    put(key, value) {

    }
